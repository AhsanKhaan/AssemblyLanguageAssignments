.MODEL SMALL
.STACK 200h
.DATA
arr DW 201,-43,54,-67,-87,87,32,-43,53,-909,231,324,-54
NGT DB 13,10,"NEGATIVE NUMBERS:$"
POS DB "POSITIVE NUMBERS:$"
TEMP DW 0  
P DB 0
N DB 0
.CODE

MAIN PROC

MOV AX,@DATA
MOV DS,AX

MOV SI,0 
MOV CX,0
MOV BX,0

L1:
CMP SI,26
JE PRINT
MOV AX,0
MOV AX,ARR[SI]
SHL AX,1
JC NEGITIVE
INC P
ADD SI,2
JMP L1 

NEGITIVE:
INC N
ADD SI,2
JMP L1 

PRINT:

MOV AH,9
MOV DX,OFFSET POS
INT 21h

MOV AH,2
MOV DL,13
INT 21h

MOV DL,10
INT 21h
MOV DX,0

MOV AX,0
MOV AL,P
CALL OUTDEC
 

MOV AH,9
MOV DX,OFFSET NGT
INT 21h

MOV AH,2
MOV DL,13
INT 21h

MOV DL,10
INT 21h
MOV DX,0

MOV AX,0
MOV AL,N
CALL OUTDEC    

MOV AH,4CH
INT 21H

MAIN ENDP 

OUTDEC PROC 

PUSH AX
PUSH BX 
PUSH DX
PUSH CX 

MOV CX,0
CMP AX,0
JGE @ELSE

PUSH AX 

MOV AH,2
MOV DL,'-'
INT 21h

POP AX
NEG  AX 
@ELSE:


MOV DX,0
MOV  BX,10
DIV BX 
PUSH DX 
INC CX 
CMP AX,0
JNZ @ELSE

MOV AH,2

PRINT_LABEL:
POP DX 
ADD DL,48
INT 21h

LOOP PRINT_LABEL 
POP DX 
POP CX 
POP BX 
POP AX 

RET 

OUTDEC ENDP 

END MAIN
